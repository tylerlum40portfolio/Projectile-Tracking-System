87static uint16_t Controller_ReadADC(uint32_t channel)
88{
89 ADC_ChannelConfTypeDef sConfig = {0};
90 uint16_t adc_value = 0;
91
92 sConfig.Channel = channel;
93 sConfig.Rank = 1;
94 sConfig.SamplingTime = ADC_SAMPLETIME_84CYCLES;
95
96 if (HAL_ADC_ConfigChannel(&hadc1, &sConfig) != HAL_OK)
97 {
98 return 0;
99 }
100
101 HAL_ADC_Start(&hadc1);
102
103 if (HAL_ADC_PollForConversion(&hadc1, 10) == HAL_OK)
104 {
105 adc_value = (uint16_t)HAL_ADC_GetValue(&hadc1);
106 }
107
108 HAL_ADC_Stop(&hadc1);
109
110 return adc_value;
111}

123// TILT HOMING

125uint8_t Controller_TiltIsHome(void)
126{
127 uint16_t tilt_adc = Controller_ReadTiltADC();
128
129 // Your measured vertical center was about 3856.
130 // This accepts 3820 to 3890 as close enough to vertical.
131 if (tilt_adc >= 3820 && tilt_adc <= 3890)
132 {
133 return 1u;
134 }
135
136 return 0u;
137}

139void Controller_HomeTiltStep(void)
140{
141 uint16_t tilt_adc = Controller_ReadTiltADC();
142
143 // If tilt ADC is above vertical center, move back toward center.
144 if (tilt_adc > 3890)
145 {
146 TiltMotor_Set(600);
147 }
148
149 // If tilt ADC is below vertical center, move the opposite direction.
150 else if (tilt_adc < 3820)
151 {
152 TiltMotor_Set(-600);
153 }
154
155 // If close to vertical, stop.
156 else
157 {
158 TiltMotor_Set(0);
159 }
160}

162void Controller_HomeTiltBlocking(uint32_t timeout_ms)
163{
164 uint32_t start_time = HAL_GetTick();
165
166 while (Controller_TiltIsHome() == 0u)
167 {
168 Controller_HomeTiltStep();
169
170 // Important: pan should NOT move while tilt is homing.
171 PanMotor_Set(0);
172 Laser_Off();
173
174 if ((HAL_GetTick() - start_time) > timeout_ms)
175 {
176 break;
177 }
178
179 HAL_Delay(20);
180 }
181
182 Motors_Stop();
183}

222// SEARCH MODE
223
224void Controller_SearchMode(void)
225{
226 uint16_t pan_adc = Controller_ReadPanADC();
227
228 Laser_Off();
229
230 // Do NOT pan until tilt is back at vertical home.
231 if (Controller_TiltIsHome() == 0u)
232 {
233 PanMotor_Set(0);
234 Controller_HomeTiltStep();
235 return;
236 }
237
238 // Once tilt is vertical, then pan can search.
239 if (pan_adc >= 3150 && pan_adc <= 3220)
240 {
241 pan_search_direction = -1;
242 }
243 else if (pan_adc >= 2550 && pan_adc <= 2750)
244 {
245 pan_search_direction = 1;
246 }
247
248 PanMotor_Set(pan_search_direction * 700);
249}

251// OPTIONAL FULL CONTROLLER UPDATE
252void Controller_Update(PixyTarget_t *camera_target, Lidar_t *lidar)
253{
254 bool target_found = Camera_FindTarget(camera_target);
255
256 if (!target_found)
257 {
258 PanMotor_Set(0);
259 Laser_Off();
260 Controller_SearchMode();
261 Controller_Reset();
262 return;
263 }
264
265 float pan_speed = PID_Update(&pan_pid, (float)camera_target->x_error);
266 float tilt_speed = PID_Update(&tilt_pid, (float)camera_target->y_error);
267
268 int16_t pan_cmd = Controller_ApplyPanLimits((int16_t)pan_speed);
269 int16_t tilt_cmd = Controller_ApplyTiltLimits((int16_t)tilt_speed);
270
271 PanMotor_Set(pan_cmd);
272 TiltMotor_Set(tilt_cmd);
273
274 LidarStatus_t lidar_status = Lidar_ReadFrame(lidar, 10u);
275
276 bool centered = Camera_TargetCentered(camera_target);
277 bool in_range = (lidar_status == LIDAR_OK) &&
278 Lidar_ObjectDetected(lidar, LASER_MAX_RANGE_CM);
279
280 if (centered && in_range)
281 {
282 Laser_On();
283 }
284 else
285 {
286 Laser_Off();
287 }
288}

292static void PID_Init(PID_t *pid, float kp, float ki, float kd,
293 float integral_max, float output_max)
294{
295 pid->kp = kp;
296 pid->ki = ki;
297 pid->kd = kd;
298 pid->integral = 0.0f;
299 pid->prev_error = 0.0f;
300 pid->integral_max = integral_max;
301 pid->output_max = output_max;
302}
303
304static float PID_Update(PID_t *pid, float error)
305{
306 pid->integral += error;
307
308 if (pid->integral > pid->integral_max)
309 {
310 pid->integral = pid->integral_max;
311 }
312 else if (pid->integral < -pid->integral_max)
313 {
314 pid->integral = -pid->integral_max;
315 }
316
317 float derivative = error - pid->prev_error;
318 pid->prev_error = error;
319
320 float output = (pid->kp * error) +
321 (pid->ki * pid->integral) +
322 (pid->kd * derivative);
323
324 if (output > pid->output_max)
325 {
326 output = pid->output_max;
327 }
328 else if (output < -pid->output_max)
329 {
330 output = -pid->output_max;
331 }
332
333 return output;
334}
