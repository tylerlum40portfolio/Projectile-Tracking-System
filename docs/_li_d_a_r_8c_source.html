117LidarStatus_t Lidar_ReadFrame(Lidar_t *dev, uint32_t timeout_ms)
118{
119 if ((dev == NULL) || (dev->huart == NULL))
120 {
121 return LIDAR_NULL_ARG;
122 }
123
124 dev->valid = 0u;
125
126 uint8_t byte = 0u;
127 uint8_t frame[LIDAR_FRAME_LEN] = {0u};
128 uint8_t index = 0u;
129 uint8_t state = 0u;
130 uint8_t saw_bad_checksum = 0u;
131
132 uint32_t start = HAL_GetTick();
133
134 while (Lidar_ElapsedMs(start) < timeout_ms)
135 {
136 HAL_StatusTypeDef hal_status;
137
138 hal_status = HAL_UART_Receive(dev->huart, &byte, 1u, 1u);
139
140 if (hal_status == HAL_TIMEOUT)
141 {
142 continue;
143 }
144
145 if (hal_status != HAL_OK)
146 {
147 return LIDAR_UART_ERROR;
148 }
149
150 switch (state)
151 {
152 case 0u:
153 if (byte == LIDAR_HEADER)
154 {
155 frame[0] = byte;
156 state = 1u;
157 }
158 break;
159
160 case 1u:
161 if (byte == LIDAR_HEADER)
162 {
163 frame[1] = byte;
164 index = 2u;
165 state = 2u;
166 }
167 else
168 {
169 state = 0u;
170 }
171 break;
172
173 case 2u:
174 frame[index] = byte;
175 index++;
176
177 if (index >= LIDAR_FRAME_LEN)
178 {
179 uint8_t checksum = 0u;
180
181 for (uint8_t i = 0u; i < LIDAR_FRAME_LEN - 1u; i++)
182 {
183 checksum = (uint8_t)(checksum + frame[i]);
184 }
185
186 if (checksum != frame[8])
187 {
188 saw_bad_checksum = 1u;
189 state = 0u;
190 index = 0u;
191 break;
192 }
193
194 dev->distance_cm = (uint16_t)frame[2] |
195 ((uint16_t)frame[3] << 8);
196
197 dev->strength = (uint16_t)frame[4] |
198 ((uint16_t)frame[5] << 8);
199
200 dev->temperature_raw = (uint16_t)frame[6] |
201 ((uint16_t)frame[7] << 8);
202
203 dev->temperature_c = ((float)dev->temperature_raw / 8.0f) - 256.0f;
204
205 dev->valid = Lidar_IsMeasurementValid(dev);
206
207 return LIDAR_OK;
208 }
209 break;
210
211 default:
212 state = 0u;
213 index = 0u;
214 break;
215 }
216 }
217
218 if (saw_bad_checksum)
219 {
220 return LIDAR_BAD_CHECKSUM;
221 }
222
223 return LIDAR_TIMEOUT;
224}
