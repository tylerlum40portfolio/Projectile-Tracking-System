147static void PID_Init(PID_t *pid, float kp, float ki, float kd,
148 float integral_max, float output_max)
149{
150 pid->kp = kp;
151 pid->ki = ki;
152 pid->kd = kd;
153 pid->integral = 0.0f;
154 pid->prev_error = 0.0f;
155 pid->integral_max = integral_max;
156 pid->output_max = output_max;
157}
158
171static float PID_Update(PID_t *pid, float error)
172{
173 /* Integral with anti-windup */
174 pid->integral += error;
175 if (pid->integral > pid->integral_max) { pid->integral = pid->integral_max; }
176 else if (pid->integral < -pid->integral_max) { pid->integral = -pid->integral_max; }
177
178 /* Derivative */
179 float derivative = error - pid->prev_error;
180 pid->prev_error = error;
181
182 /* PID sum */
183 float output = (pid->kp * error)
184 + (pid->ki * pid->integral)
185 + (pid->kd * derivative);
186
187 /* Output clamp */
188 if (output > pid->output_max) { output = pid->output_max; }
189 else if (output < -pid->output_max) { output = -pid->output_max; }
190
191 return output;
192}
