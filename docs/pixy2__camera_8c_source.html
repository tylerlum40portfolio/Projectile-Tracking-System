97static int16_t Pixy2_FindResponseStart(const uint8_t *buffer, uint16_t length)
98{
99 for (uint16_t i = 0; i < length - 1; i++)
100 {
101 if (buffer[i] == PIXY2_RESP_SYNC_1 && buffer[i + 1] == PIXY2_RESP_SYNC_2)
102 {
103 return (int16_t)i;
104 }
105 }
106
107 return -1;
108}

110/*
111 * Low-level Pixy2 request.
112 * Sends request packet, then clocks in response bytes using dummy bytes.
113 */
126static HAL_StatusTypeDef Pixy2_SendRequest(const uint8_t *request,
127 uint16_t request_len,
128 uint8_t *response,
129 uint16_t response_len)
130{
131 HAL_StatusTypeDef status;
132 uint8_t dummy_tx[PIXY2_RESPONSE_BUFFER_SIZE];
133
134 if (response_len > PIXY2_RESPONSE_BUFFER_SIZE)
135 {
136 response_len = PIXY2_RESPONSE_BUFFER_SIZE;
137 }
138
139 memset(dummy_tx, 0x00, sizeof(dummy_tx));
140 memset(response, 0x00, response_len);
141
142 Pixy2_Select();
143
144 status = HAL_SPI_Transmit(&hspi1,
145 (uint8_t *)request,
146 request_len,
147 PIXY2_SPI_TIMEOUT_MS);
148
149 if (status == HAL_OK)
150 {
151 /*
152 * Pixy2 sometimes needs a very small delay before response bytes.
153 */
154 HAL_Delay(1);
155
156 status = HAL_SPI_TransmitReceive(&hspi1,
157 dummy_tx,
158 response,
159 response_len,
160 PIXY2_SPI_TIMEOUT_MS);
161 }
162
163 Pixy2_Deselect();
164
165 return status;
166}

180/*
181 * Reads Pixy2 CCC blocks.
182 *
183 * Returns:
184 * number of valid blocks copied into blocks[]
185 *
186 * For your test:
187 * PIXY2_TARGET_SIGNATURE = 1
188 * That means sigmap = 0x01, so Pixy2 only returns signature 1 blocks.
189 */
201uint8_t getCameraBlocks(PixyBlock_t *blocks, uint8_t max_blocks)
202{
203 uint8_t request[6];
204 uint8_t response_raw[PIXY2_RESPONSE_BUFFER_SIZE];
205
206 if (blocks == NULL || max_blocks == 0)
207 {
208 return 0;
209 }
210
211 if (max_blocks > PIXY2_MAX_BLOCKS)
212 {
213 max_blocks = PIXY2_MAX_BLOCKS;
214 }
215
216 /*
217 * Sigmap controls which signatures Pixy2 returns.
218 *
219 * Signature 1 -> bit 0 -> 0x01
220 * Signature 2 -> bit 1 -> 0x02
221 * Signature 3 -> bit 2 -> 0x04
222 *
223 * Since your red target is Signature 1, sigmap = 0x01.
224 */
225 uint8_t sigmap = (uint8_t)(1 << (PIXY2_TARGET_SIGNATURE - 1));
226
227 /*
228 * getBlocks(sigmap, maxBlocks)
229 *
230 * Byte 0: 0xAE
231 * Byte 1: 0xC1
232 * Byte 2: 0x20, getBlocks request
233 * Byte 3: 0x02, payload length
234 * Byte 4: sigmap
235 * Byte 5: max blocks
236 */
237 request[0] = PIXY2_REQ_SYNC_1;
238 request[1] = PIXY2_REQ_SYNC_2;
239 request[2] = PIXY2_TYPE_GET_BLOCKS_REQ;
240 request[3] = 0x02;
241 request[4] = sigmap;
242 request[5] = max_blocks;
243
244 HAL_StatusTypeDef status = Pixy2_SendRequest(request,
245 sizeof(request),
246 response_raw,
247 sizeof(response_raw));
248
249 if (status != HAL_OK)
250 {
251 return 0;
252 }
253
254 int16_t start = Pixy2_FindResponseStart(response_raw, sizeof(response_raw));
255
256 if (start < 0)
257 {
258 return 0;
259 }
260
261 uint8_t *response = &response_raw[start];
262
263 /*
264 * Response format:
265 * [0] = 0xAF
266 * [1] = 0xC1
267 * [2] = packet type, should be 0x21
268 * [3] = payload length
269 * [4] = checksum low
270 * [5] = checksum high
271 * [6...] = block data
272 */
273 if ((start + 6) >= PIXY2_RESPONSE_BUFFER_SIZE)
274 {
275 return 0;
276 }
277
278 if (response[0] != PIXY2_RESP_SYNC_1 ||
279 response[1] != PIXY2_RESP_SYNC_2)
280 {
281 return 0;
282 }
283
284 if (response[2] != PIXY2_TYPE_GET_BLOCKS_RESP)
285 {
286 return 0;
287 }
288
289 uint8_t payload_len = response[3];
290
291 if (payload_len < PIXY2_BLOCK_SIZE)
292 {
293 return 0;
294 }
295
296 if ((start + 6 + payload_len) > PIXY2_RESPONSE_BUFFER_SIZE)
297 {
298 return 0;
299 }
300
301 uint16_t received_checksum = Pixy2_ReadU16(&response[4]);
302 uint16_t calculated_checksum = Pixy2_Checksum(&response[6], payload_len);
303
304 if (received_checksum != calculated_checksum)
305 {
306 return 0;
307 }
308
309 uint8_t available_blocks = payload_len / PIXY2_BLOCK_SIZE;
310
311 if (available_blocks > max_blocks)
312 {
313 available_blocks = max_blocks;
314 }
315
316 for (uint8_t i = 0; i < available_blocks; i++)
317 {
318 uint8_t *block = &response[6 + (i * PIXY2_BLOCK_SIZE)];
319
320 blocks[i].signature = Pixy2_ReadU16(&block[0]);
321 blocks[i].x = Pixy2_ReadU16(&block[2]);
322 blocks[i].y = Pixy2_ReadU16(&block[4]);
323 blocks[i].width = Pixy2_ReadU16(&block[6]);
324 blocks[i].height = Pixy2_ReadU16(&block[8]);
325 blocks[i].angle = Pixy2_ReadS16(&block[10]);
326 blocks[i].index = block[12];
327 blocks[i].age = block[13];
328 }
329
330 return available_blocks;
331}

333/* Finds the red target from Pixy2 blocks, since red object is Signature 1,
334this searches for: block.signature == PIXY2_TARGET_SIGNATURE
335 If multiple red blocks are visible, it chooses the largest one by area.*/
345bool Camera_FindTarget(PixyTarget_t *target)
346{
347 PixyBlock_t blocks[PIXY2_MAX_BLOCKS];
348 uint8_t block_count;
349 uint8_t best_index = 0;
350 uint32_t best_area = 0;
351 bool found = false;
352
353 if (target == NULL)
354 {
355 return false;
356 }
357
358 target->target_detected = false;
359 target->x = 0;
360 target->y = 0;
361 target->width = 0;
362 target->height = 0;
363 target->x_error = 0;
364 target->y_error = 0;
365
366 block_count = getCameraBlocks(blocks, PIXY2_MAX_BLOCKS);
367
368 if (block_count == 0)
369 {
370 return false;
371 }
372
373 for (uint8_t i = 0; i < block_count; i++)
374 {
375 if (blocks[i].signature == PIXY2_TARGET_SIGNATURE)
376 {
377 uint32_t area = (uint32_t)blocks[i].width * (uint32_t)blocks[i].height;
378
379 if (area < PIXY2_MIN_TARGET_AREA) {
380 continue;
381 }
382
383 if (!found || area > best_area)
384 {
385 found = true;
386 best_area = area;
387 best_index = i;
388 }
389 }
390 }
391
392 if (!found)
393 {
394 return false;
395 }
396
397 target->target_detected = true;
398 target->x = blocks[best_index].x;
399 target->y = blocks[best_index].y;
400 target->width = blocks[best_index].width;
401 target->height = blocks[best_index].height;
402
403 target->x_error = (int16_t)target->x - PIXY2_CENTER_X;
404 target->y_error = (int16_t)target->y - PIXY2_CENTER_Y;
405
406 return true;
407}
